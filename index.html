function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Attendance"); const data = JSON.parse(e.postData.contents); const name = data.name; const telegramId = data.telegramId; const type = data.type; const lat = data.lat; const lng = data.lng; const photo = data.photo; let photoUrl = "no-photo"; // 📸 Upload photo to Google Drive if(photo && photo.startsWith("data:image")){ const base64Data = photo.split(",")[1]; const blob = Utilities.newBlob(Utilities.base64Decode(base64Data), "image/png", "photo_"+Date.now()+".png"); const folder = DriveApp.getFolderById("1wCs2u3r3jmVnkXxNPFooqevYAX4Xa9PG"); // 🔁 ដាក់ Folder ID const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); photoUrl = file.getUrl(); } // 🛑 Prevent duplicate (same day + same type) const lastRow = sheet.getLastRow(); const values = sheet.getRange(2,1,lastRow-1,8).getValues(); const today = new Date().toDateString(); for(let i = values.length -1; i >=0; i--){ let row = values[i]; let rowDate = new Date(row[6]).toDateString(); if(row[1] == telegramId && rowDate == today && row[2] == type){ return ContentService .createTextOutput(JSON.stringify({status:"error", message:"Already checked"})) .setMimeType(ContentService.MimeType.JSON); } } // ✅ Save to Sheet sheet.appendRow([ name, telegramId, type, lat, lng, photoUrl, new Date(), "OK" ]); return ContentService .createTextOutput(JSON.stringify({ status:"success", photo: photoUrl })) .setMimeType(ContentService.MimeType.JSON); }
